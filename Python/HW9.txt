def turn_atom(atom):
    return (atom[0], (atom[1] + 1) % 4)

def metamap(f, listOfRows):
    return [list(map(f, row)) for row in listOfRows]

print("Testing turn_atom()")
a = ('a', 0)
for i in range(5):
    print(a)
    a = turn_atom(a)

print("\nTesting metamap()")
matrix = [[1,2,3], [4,5,6], [7,8,9]]
newMatrix = metamap(lambda x: x*x, matrix)
for row in matrix:
    print (row)
