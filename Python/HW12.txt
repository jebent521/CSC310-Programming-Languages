from functools import wraps
def printit(f):
    @wraps(f)
    def wrapper(*args, **kwargs):
        arguments = ''
        for arg in args:
            arguments += (str(arg) + ', ')
        for k, v in kwargs.items():
            arguments += ('='.join((k, str(v))) + ', ')
        print(f.__name__, 'input:', arguments[:-2])
        print('output:', f(*args, **kwargs))
        return f(*args, **kwargs)
    return wrapper
